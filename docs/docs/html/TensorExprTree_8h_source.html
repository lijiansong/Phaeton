100 #define GEN_EXPR_NODE_CLASS_DECL(ExprName) \
101  class ExprName##Expr : public ExpressionNode { \
102  public: \
103  ExprName##Expr(ExpressionNode *LHS, ExpressionNode *RHS) \
104  : ExpressionNode(EXPR_KIND_##ExprName, 2) { \
105  setChild(0, LHS); \
106  setChild(1, RHS); \
107  /* Type checking for element-wise expr, the following must hold: \

109  */ \
110  setDims(LHS->getDims()); \
111  } \
112  \

114  virtual void transform(ExprTreeTransformer *Transformer); \
115  \
116  static ExprName##Expr *create(ExpressionNode *LHS, ExpressionNode *RHS) { \
117  return new ExprName##Expr(LHS, RHS); \
118  } \
119  };
