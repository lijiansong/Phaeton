99 #define GEN_EXPR_NODE_CLASS_DECL(ExprName) \
100  class ExprName##Expr : public ExprNode { \
101  public: \
102  ExprName##Expr(ExprNode *LHS, ExprNode *RHS) \
103  : ExprNode(EXPR_KIND_##ExprName, 2) { \
104  setChild(0, LHS); \
105  setChild(1, RHS); \
106  /* Type checking for element-wise expr, the following must hold: \
107  * LHS->getDims() == RHS->getDims() \
108  */ \
109  setDims(LHS->getDims()); \
110  } \
111  \
112  virtual void visit(ExprTreeVisitor *Visitor) const; \
113  virtual void transform(ExprTreeTransformer *Transformer); \
114  \
115  static ExprName##Expr *create(ExprNode *LHS, ExprNode *RHS) { \
116  return new ExprName##Expr(LHS, RHS); \
117  } \
118  };
